import sys

def func_position(a, b):
    print('%s: a: %s, b: %s' % (sys._getframe().f_code.co_name, a, b))

def func_default(a, b=10):
    print('%s: a: %s, b: %s' % (sys._getframe().f_code.co_name, a, b))

def func_available(a, *args):
    print('%s: a: %s, args is a tuple name: %s' % (sys._getframe().f_code.co_name, a, args))

# x can not be the same name with in any of kw
def func_key(x, **kw):
    print('%s: a: %s, kw is a dict name: %s' % (sys._getframe().f_code.co_name, x, kw))

def func_named_key_no_available(a, *, name, city):
    print('%s: a: %s, name: %s, city: %s' % (sys._getframe().f_code.co_name, a, name, city))

def func_named_key_with_available(a, *args, name, city):
    print('%s: a: %s, args: %s, name: %s, city: %s' % (sys._getframe().f_code.co_name, a, args, name, city))


if __name__ == '__main__':

def fact(x):

#!/usr/bin/env python3

def iter_num():
    n = 1
    while True:
        n = n + 1
        yield n

def not_division(n):
    return lambda x: x % n > 0

def primes():
    curiter = iter_num()
    while True:
        firstnum = next(curiter)
        yield firstnum
        curiter = filter(not_division(firstnum), curiter)

def max_prime(max):
    # max: max prime value
    #for n in primes():
    #    if n < max:
    #        print(n)
    #    else:
    #        break

    # max: number of prime value
    n = 0
    p = primes()
    while n < max:
        print(next(p))
        n = n + 1

if __name__ == '__main__':

def lazy_xx0():
    L = []
    for i in range(1, 4):
        def ifunc():
            return i * i
        L.append(ifunc)
    return L

decorator: improve function

#!/usr/bin/env python3

import functools

### case 1: if no parameter in decorator function {
def mydecorator1(func):
    # XXX: `@functools.wraps(func)` to fix wrong `__name__` in func
    @functools.wraps(func)
    def wrapper(*args, **kw):
        # do something more
        return func(*args, **kw)
    return wrapper

@mydecorator1
def test1():
    print('in test1')

# `@mydecorator1` is same with: test1 = mydecorator1(test1)
# then `test1` is the same with `wrapper`
# and old test1 function will be called as `func` in `wrapper` function
# now if you call test1(), it will call wrapper()
# } end

### case 2: if with parameter in decorator function {
def outerfunc(more_info):
    def mydecorator2(func):
        # XXX: `@functools.wraps(func)` to fix wrong `__name__` in func
        @functools.wraps(func)
        def wrapper(*args, **kw):
            print('parameter: %s' % more_info)
            return func(*args, **kw)
        return wrapper
    return mydecorator2

@outerfunc('xxx')
def test2():
    print('in test2')

# `@outerfunc('xxx')` is same with: test2 = outerfunc('xxx')(test2)
# 1: exec outerfunc('xxx'), then return mydecorator2
# 2: exec mydecorator2(test2), then return wrapper to new test2, it is same with: test2 = mydecorator2(test2)
# 3: now if you call test2(), it will call wrapper()
# } end

# **the last returned function will be wrapper, so call `@functools.wraps(func)` on `wrapper`**
