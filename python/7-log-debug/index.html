unittest

define your cases

import unittest
from your_package import YourClass

class TestYourClass(unittest.TestCase):
    def setUp(self):
        pass
    def tearDown(self):
        pass
    def test_xxx(self):
        do something
        self.assertEqual()
        self.assertTrue()
        with self.assertRaises(AttributeError):
            do something
    def test_yyy(self):
        pass

if __name__ == '__main__':
    unittest.main()

run cases

  • python3 a.py
  • python3 -m unittest a.py
