show gif

  • add gif file to resource
  • add a Label in a dialog (for example)
  • #include <QMovie>
  • add private data member QMovie *movie;
  • add 2 buttons
  • set movie in label, connect 2 signal/slot(s)
#include "aboutdialog.h"
#include "ui_aboutdialog.h"

AboutDialog::AboutDialog(QWidget *parent) :
    QDialog(parent),
    ui(new Ui::AboutDialog)
{
    ui->setupUi(this);
    movie = new QMovie(":/resource/icons/resource/play.gif");
    ui->labelMovie->setMovie(movie);
    connect(ui->pushButtonStart, SIGNAL(clicked()), movie, SLOT(start()));
    connect(ui->pushButtonStop, SIGNAL(clicked()), movie, SLOT(stop()));
    movie->start();
}

AboutDialog::~AboutDialog()
{
    delete movie;
    delete ui;
}
